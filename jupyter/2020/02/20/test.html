import io
import numpy as np
import png
def png_bytes(a):
    output = io.BytesIO()
    b = (1 - a)
    png.from_array(b.astype(np.uint8)*255, 'L').write(output)
    return output.getvalue()

import base64
dataurl = b'data:image/png;base64,'+base64.b64encode(png_bytes(np.identity(128)))
